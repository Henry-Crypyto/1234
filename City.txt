<?php
// 设置响应的内容类型为 JSON 和字符编码为 UTF-8
header('Content-Type: application/json;charset=utf-8');

// 数据库配置
$host = "localhost";
$user = "root";
$password = "";
$database = "city";

// 连接数据库
$link = mysqli_connect($host, $user, $password) or die("数据库连接失败");

// 选择数据库
mysqli_select_db($link, $database) or die("数据库选择失败");

// 设置字符集为 utf8
mysqli_query($link, "SET NAMES utf8");

// 你的其他代码，例如查询数据库并输出 JSON 数据

?>
