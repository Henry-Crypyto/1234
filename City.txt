 $('#road').change(function() {
        var road = '';
        if ($('#road').find(":selected").text() === '請選擇') {
            road = '';
        } else {
            road = $('#road').find(":selected").text();
        }
        $("#input").val($('#city').find(":selected").text() + $('#site_id').find(":selected").text() + road);
    });
